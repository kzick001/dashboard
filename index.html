async function fetchGoogleData() { const url = 'https://script.google.com/macros/s/AKfycbxKIcuEZgMfopDpAhczuHvyTAh4qJhqZw7OeWMfxqhY9BmPyIYu4o1K9lchTODYeVGSrg/exec'; try { // 'redirect: follow' is the key for GitHub/production environments const res = await fetch(url, { method: 'GET', redirect: 'follow' }); if (!res.ok) throw new Error('Network response was not ok'); const data = await res.json(); // Render Emails const emailList = document.getElementById('emailList'); emailList.innerHTML = data.emails.length > 0 ? data.emails.map(subject => `
  • ${subject}
    • `).join('') : '
  • No starred emails.
    • '; // Render Calendar const calList = document.getElementById('calendarList'); calList.innerHTML = data.calendar.length > 0 ? data.calendar.map(item => `
  • ${item.date}
    ${item.title}
    • `).join('') : '
  • No upcoming events.
    • '; } catch (error) { console.error('Google Data Error:', error); document.getElementById('emailList').innerHTML = '
  • Connection failed.
    • '; } }
